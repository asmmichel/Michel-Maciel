package auditorio;

import java.awt.AWTException;
import java.awt.Robot;
import java.awt.event.KeyEvent;
import java.util.*;

public class Auditorio {

    public static void main(String[] args) {
        Scanner ent = new Scanner(System.in);
        Dados minhaHashtable = new Dados();
        Evento meuEvento;
        int resp;

        do {
            System.out.println("Agendamento de Auditório em Eventos");
            System.out.println("___________________________________");
            System.out.println("1 - Inserir Evento no Auditório");
            System.out.println("2 - Listar todos os Eventos Inseridos");
            System.out.println("3 - Pesquisar um Evento Inserido");
            System.out.println("4 - Editar um Evento");
            System.out.println("5 - Excluir um Evento Inserido");
            System.out.println("6 - Apagar Todos os Eventos");
            System.out.println("0 - Sair do Programa");
            System.out.print("Escolha uma Opção: ");
            resp = Integer.parseInt(ent.nextLine());

            if (resp == 1) {
                ConsoleClear();
                meuEvento = new Evento();
                meuEvento.inserirEvento();
                minhaHashtable.gravarDados(meuEvento);
                System.out.println("Evento Inserido! - Número de Eventos: "
                        + minhaHashtable.hashEvento.size());
                ent.nextLine();
            }
            if (resp == 2) {
                ConsoleClear();
                if (minhaHashtable.hashEvento.isEmpty()) {
                    System.out.println("Não Existem Eventos no Auditório");
                } else {
                    minhaHashtable.listarEventos();
                    System.out.println("Número de Eventos: "
                            + minhaHashtable.hashEvento.size());
                }
                ent.nextLine();
            }
            if (resp == 3) {
                ConsoleClear();
                if (minhaHashtable.hashEvento.isEmpty()) {
                    System.out.println("Não Existem Eventos no Auditório");
                } else {
                    System.out.print("Código do Evento: ");
                    String aux = ent.nextLine();
                    minhaHashtable.pesquisarEvento(aux);
                }
                ent.nextLine();
            }
            if (resp == 4) {
                ConsoleClear();
                if (minhaHashtable.hashEvento.isEmpty()) {
                    System.out.println("Não Existem Eventos no Auditório");
                } else {
                    System.out.print("Código do Evento: ");
                    String aux = ent.nextLine();
                    minhaHashtable.editarEvento(aux);
                }
                ent.nextLine();
            }
            if (resp == 5) {
                ConsoleClear();
                if (minhaHashtable.hashEvento.isEmpty()) {
                    System.out.println("Não Existem Eventos no Auditório");
                } else {
                    System.out.print("Código do Evento: ");
                    String aux = ent.nextLine();
                    minhaHashtable.removerEvento(aux);
                    System.out.println("Número de Eventos: "
                            + minhaHashtable.hashEvento.size());
                }
                ent.nextLine();
            }
            if (resp == 6) {
                ConsoleClear();
                if (minhaHashtable.hashEvento.isEmpty()) {
                    System.out.println("Não Existem Eventos no Auditório");
                } else {
                    minhaHashtable.hashEvento.clear();
                    System.out.println("Todos os Eventos foram Apagados do Auditório!");
                    System.out.println("Número de Eventos: "
                            + minhaHashtable.hashEvento.size());
                }
                ent.nextLine();
            }
            ConsoleClear();
        } while (resp != 0);
    }
    
    static void ConsoleClear() {
        try {
            Robot robot = new Robot();

            robot.setAutoDelay(5);
            robot.keyPress(KeyEvent.VK_CONTROL);
            robot.keyPress(KeyEvent.VK_L);

            robot.keyRelease(KeyEvent.VK_CONTROL);
            robot.keyRelease(KeyEvent.VK_L);

        } catch (AWTException ex) {
            System.out.println(ex.getMessage());
        }
    }
}
