package auditorio;

import java.util.*;

public class Evento {

    public String codEvento;
    public String nomeEvento;
    public String dataEvento;
    public String horaEvento;

    public Evento() {
        UUID myUUID = UUID.randomUUID();
        this.codEvento = myUUID.toString().toUpperCase().substring(15, 18);
    }

    public void inserirEvento() {
        Scanner ent = new Scanner(System.in);
        System.out.println("Código do Evento: "+this.codEvento);
        System.out.print("Digite o Nome do Evento: ");
        this.nomeEvento = ent.nextLine();
        System.out.print("Digite a Data do Evento: ");
        this.dataEvento = ent.nextLine();
        System.out.print("Digite o Horário do Evento: ");
        this.horaEvento = ent.nextLine();
    }
}
