package seo;

class Elemento {

    public Func func;
    public Elemento prox;

    public Elemento() {
        func = null;
        prox = null;
    }
}

public class Lista {

    private Elemento inicio;
    private Elemento fim;
    private Elemento aux;
    private Elemento anterior;
    public int tamanho;

    public Lista() {

        inicio = null;
        fim = null;
        tamanho = 0;
    }

    public void inserir(Func meufunc) {
        Elemento novo = new Elemento();
        novo.func = meufunc;
        if (inicio == null) {
            inicio = novo;
            fim = novo;
        } else {
            anterior = null;
            aux = inicio;
            while (aux != null && novo.func.nome.compareTo(aux.func.nome) > 0) {
                anterior = aux;
                aux = aux.prox;
            }
            if (anterior == null) {
                novo.prox = inicio;
                inicio = novo;
            } else if (aux == null) {
                fim.prox = novo;
                fim = novo;
            } else {
                anterior.prox = novo;
                novo.prox = aux;
            }
        }
        tamanho++;
    }
    
    public void mostrar() {
        if (inicio == null) {
            System.out.println("N達o existem Dados no Cadastro");
        } else {
            aux = inicio;
            while (aux != null) {
                System.out.println(aux.func.nome + " - " + aux.func.matricula+"");
                aux = aux.prox;
            }
            System.out.println("\nTotal de Registros: " + tamanho);
        }
    }

    public void retirar(String meufunc) {
        if (inicio == null) {
            System.out.println("N達o existem Dados no Cadastro");
        } else {
            boolean achou = false;
            anterior = null;
            aux = inicio;                                               
            while (aux != null) {
                if (aux.func.matricula.toUpperCase().compareTo
                    (meufunc.toUpperCase()) == 0) {
                    achou = true;
                    if (aux == inicio) {
                        inicio = inicio.prox;              
                    } else if (aux == fim) {
                        anterior.prox = null;           
                        fim = anterior;
                    } else {
                        anterior.prox = aux.prox;       
                    }
                    tamanho--;
                } else {
                    anterior = aux;
                    aux = aux.prox;
                }
            }
            if (!achou) {
                System.out.println("Funcionario n達o cadastrado");
            } else {
                System.out.println("Funcionario Removido do Cadastro");
            }
            System.out.println("Total de Registros: " + tamanho);
        }
    }

    public void esvaziar() {
        if (inicio == null) {
            System.out.println("N達o existem Dados no Cadastros");
        } else {
            inicio = null;
            tamanho = 0;
            System.out.println("Os Dados do Cadastro foram removidos");
        }
    }
}
