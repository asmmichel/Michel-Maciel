package deo;

import java.awt.AWTException;
import java.awt.Robot;
import java.awt.event.KeyEvent;
import java.util.Scanner;

public class Deo {

    public static void main(String[] args) {
        Scanner ent = new Scanner(System.in);

        Lista minhalista = new Lista();
        int num, r;

        do {
            System.out.println("Lista DEO");
            System.out.println("1 - Inserir Número");
            System.out.println("2 - Mostrar Número");
            System.out.println("3 - Remover Número");
            System.out.println("4 - Apagar Tudo");
            System.out.println("0 - Sair");
            System.out.print("Escolha uma opção: ");
            r = Integer.parseInt(ent.nextLine());

            if (r == 1) {
                ConsoleClear();
                System.out.print("Digite um Número - ");
                num = Integer.parseInt(ent.nextLine());
                minhalista.inserir(num);
                System.out.println("Número Inserido, Registros: " 
                        + minhalista.tamanho);
                ent.nextLine();
            }
            if (r == 2) {
                ConsoleClear();
                minhalista.mostrar();
                ent.nextLine();
            }
            if (r == 3) {
                ConsoleClear();
                System.out.print("Digite o Número para Remover - ");
                num = Integer.parseInt(ent.nextLine());
                minhalista.remover(num);
                ent.nextLine();
            }
            if (r == 4) {
                ConsoleClear();
                minhalista.esvaziar();
                ent.nextLine();
            }
            ConsoleClear();
        } while (r != 0);
    }
    
        static void ConsoleClear() {
        try {
            Robot robot = new Robot();

            robot.setAutoDelay(5);
            robot.keyPress(KeyEvent.VK_CONTROL);
            robot.keyPress(KeyEvent.VK_L);

            robot.keyRelease(KeyEvent.VK_CONTROL);
            robot.keyRelease(KeyEvent.VK_L);

        } catch (AWTException ex) {
            System.out.println(ex.getMessage());
        }
    }
}
