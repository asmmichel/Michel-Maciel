package arec;

import java.util.*;

public class Arec {

    static int ContarInverso(int n, int nUM) {

        int soma;

        if (n == 1) {
            soma = 1;
            System.out.print(n + " ");
        } else {
            soma = n * nUM + ContarInverso(n - 1, -nUM);
            System.out.print(n + " ");
        }
        return soma;
    }

    public static void main(String[] args) {

        Scanner ent = new Scanner(System.in);

        int n, soma;

        System.out.print("Digite um NÃºmero - ");
        n = ent.nextInt();

        soma = ContarInverso(n, 1);

        System.out.println("\nSoma Inversa = " + soma);
    }

}
