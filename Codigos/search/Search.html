package search;

import java.util.Scanner;

public class Search {

    public static void main(String[] args) {
        Scanner ent = new Scanner(System.in);
        
        int num;
        int inicio=0;
        int fim=999999;
        int meio=(inicio+fim)/2;
        int[] vetor = new int[1000000];
        boolean achou=false;
        
        System.out.print("Velocidade da Pesquisa 1:1000000 - ");
        num = ent.nextInt();
        
        for (int i = 0; i < 1000000; i++) {
            vetor[i] = i;
        }
        
        while (inicio <= fim && !achou) {
            if (vetor[meio]==num) {
                achou=true;
            } else {
                if (num<vetor[meio]) {
                    fim=meio-1;
                } else {
                    inicio=meio+1;
                }
                meio=(inicio+fim)/2;
            }
        }
        if (!achou) {
            System.out.println("Não Encontrado");
        } else {
            System.out.println("Encontrado na posição "+(meio+1)+" do vetor");
        }
    }

}
