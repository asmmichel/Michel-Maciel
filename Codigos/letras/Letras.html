package letras;

import java.util.*;

public class Letras {

    static int calcTamanho(String frase) {

        if (frase.isEmpty()) {
            return 0;
        } else {
            frase = frase.substring(1);
            return 1 + calcTamanho(frase);
        }
    }

    public static void main(String[] args) {
        Scanner ent = new Scanner(System.in);

        String frase;
        int qLetras;

        System.out.print("Frase = ");
        frase = ent.nextLine();

        qLetras = calcTamanho(frase);

        System.out.println("A frase tem " + qLetras + " letras");
    }

}
