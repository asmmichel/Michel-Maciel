package fila;

import java.awt.AWTException;
import java.awt.Robot;
import java.awt.event.KeyEvent;
import java.util.*;

public class Fila {

    public static void main(String[] args) {
        Scanner ent = new Scanner(System.in);

        FilaCarro minhafila = new FilaCarro();
        int r;

        do {
            System.out.println("Menu de Opções");
            System.out.println("1 - Entrada de Veículo no Pátio");
            System.out.println("2 - Saída de Veículo do Pátio");
            System.out.println("3 - Listagem Geral de Veículos");
            System.out.println("4 - Esvaziar Pátio");
            System.out.println("0 - Sair do Programa");
            System.out.print("Escolha uma Opção: ");
            r = Integer.parseInt(ent.nextLine());
            
            if (r == 1) {
                ConsoleClear();
                if (minhafila.numcarros == 5) {
                    System.out.println("Patio Lotado");
                } else {
                    minhafila.entrada();
                    System.out.println("Registro de Entrada de Veículo");
                    System.out.println(minhafila.numcarros+" Veículo(s) no Pátio");
                }
                ent.nextLine();
            }
            if (r == 2) {
                ConsoleClear();
                if (minhafila.numcarros == 0) {
                    System.out.println("Não existem Veículos no Pátio");
                } else {
                    minhafila.retirar();
                    System.out.println("Registro de Saída de Veículo");
                    System.out.println(minhafila.numcarros+" Veículos no Pátio");
                }
                ent.nextLine();
            }
            if (r == 3) {
                ConsoleClear();
                minhafila.mostrar();
                ent.nextLine();
            }
            if (r == 4) {
                ConsoleClear();
                minhafila.esvaziar();
                System.out.println("Pátio Esvaziado");
                ent.nextLine();
            }
            ConsoleClear();
        } while (r != 0);
    }
    
        static void ConsoleClear() {
        try {
            Robot robot = new Robot();

            robot.setAutoDelay(5);
            robot.keyPress(KeyEvent.VK_CONTROL);
            robot.keyPress(KeyEvent.VK_L);

            robot.keyRelease(KeyEvent.VK_CONTROL);
            robot.keyRelease(KeyEvent.VK_L);

        } catch (AWTException ex) {
            System.out.println(ex.getMessage());
        }
    }
}
