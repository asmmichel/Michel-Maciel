package vet;

import java.util.*;

public class Vet {

    static int maior(int[] v, int mA, int i) {

        if (v[i] > mA) {
            mA = v[i];
        }

        if (i == v.length - 1) {
            return mA;
        } else {
            return maior(v, mA, i + 1);
        }
    }

    public static void main(String[] args) {

        Scanner ent = new Scanner(System.in);
        Random ale = new Random();

        int v[];
        int tamanho;
        int mA;

        System.out.print("Tamanho do Vetor = ");
        tamanho = ent.nextInt();

        v = new int[tamanho];

        for (int i = 0; i < tamanho; i++) {
            v[i] = ale.nextInt(90) + 10;
            System.out.print(v[i] + " ");
        }
        System.out.println();

        mA = maior(v, v[0], 1);

        System.out.println("Maior Elemento = " + mA);
    }

}
