package pil;

import java.awt.AWTException;
import java.awt.Robot;
import java.awt.event.KeyEvent;
import java.util.Scanner;

public class Pil {

    public static void main(String[] args) {
        Scanner ent = new Scanner(System.in);

        Pilha minhapilha = new Pilha();
        int num, r;

        do {
            System.out.println("PILHA");
            System.out.println("1 - Inserir Número");
            System.out.println("2 - Retirar Número");
            System.out.println("3 - Mostrar Número");
            System.out.println("4 - Apagar Número");
            System.out.println("0 - Sair");
            System.out.print("Escolha uma Opção: ");
            r = Integer.parseInt(ent.nextLine());

            if (r == 1) {
                ConsoleClear();
                System.out.print("Digite um Número - ");
                num = Integer.parseInt(ent.nextLine());
                minhapilha.inserir(num);
                ent.nextLine();
            }
            if (r == 2) {
                ConsoleClear();
                minhapilha.retirar();
                ent.nextLine();
            }
            if (r == 3) {
                ConsoleClear();
                minhapilha.mostrar();
                ent.nextLine();
            }
            if (r == 4) {
                ConsoleClear();
                minhapilha.apagar();
                ent.nextLine();
            }
            ConsoleClear();
        } while (r != 0);
    }
    
        static void ConsoleClear() {
        try {
            Robot robot = new Robot();

            robot.setAutoDelay(5);
            robot.keyPress(KeyEvent.VK_CONTROL);
            robot.keyPress(KeyEvent.VK_L);

            robot.keyRelease(KeyEvent.VK_CONTROL);
            robot.keyRelease(KeyEvent.VK_L);

        } catch (AWTException ex) {
            System.out.println(ex.getMessage());
        }
    }
    
}
