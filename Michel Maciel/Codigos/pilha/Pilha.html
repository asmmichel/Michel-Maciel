package pil;

class Elemento {

    public int num;
    public Elemento prox;

    public Elemento() {
        num = 0;
        prox = null;
    }
}

public class Pilha {

    private Elemento topo;
    private Elemento aux;
    public int tamanho;

    public Pilha() {
        topo = null;
        tamanho = 0;
    }

    public void inserir(int num) {

        Elemento novo = new Elemento();
        novo.num = num;

        novo.prox = topo;
        topo = novo;
        tamanho++;
        System.out.println("Inserido, Registros: " + tamanho);
    }

    public void retirar() {

        if (topo == null) {
            System.out.println("Não existem Números na Pilha");
        } else {
            topo = topo.prox;
            tamanho--;
            System.out.println("Retirado, Registros: " + tamanho);
        }
    }

    public void mostrar() {

        if (topo == null) {
            System.out.println("Não existem Números na Pilha");
        } else {
            System.out.println("Registros: " + tamanho);
            System.out.println("\tTopo");
            aux = topo;
            while (aux != null) {
                System.out.println("\t" + aux.num);
                aux = aux.prox;
            }
            System.out.println("\tBase");
        }
    }

    public void apagar() {

        if (topo == null) {
            System.out.println("Não existem Números na Pilha");
        } else {
            topo = null;
            tamanho = 0;
            System.out.println("Todos os Números Apagados, Registros: " + tamanho);
        }
    }

}
