package fila;

import java.util.*;

public class FilaCarro {

    public Queue<Carro> filacarros;
    public int numcarros;

    public FilaCarro() {
        filacarros = new LinkedList<>();
        numcarros = 0;
    }

    public void entrada() {
        Scanner ent = new Scanner(System.in);

        Carro novocarro = new Carro();
        System.out.println("Entrada de Veículos no Pátio");
        System.out.print("Placa - ");
        novocarro.placa = ent.nextLine();
        System.out.print("Nome - ");
        novocarro.nome = ent.nextLine();
        filacarros.add(novocarro);
        numcarros++;
        System.out.println();
    }

    public void mostrar() {
        
        System.out.println("Listagem de Veículos do Pátio\n");
        System.out.println("\tPortão de Saída");
        for (Carro x : filacarros) {
            System.out.println("\t"+x.placa+" - "+x.nome);
        }
        System.out.println("\tPortão de Entrada");
    }

    public void retirar() {
        Scanner ent = new Scanner(System.in);
        
        String placa;
        Carro x;
        System.out.println("Retirada de Veículo do Pátio");
        System.out.print("Placa -  ");
        placa = ent.nextLine();
        x = filacarros.peek();
        while (!x.placa.equals(placa)) {
            filacarros.remove(x);
            filacarros.add(x);
            x = filacarros.peek();
        }
        filacarros.remove();
        numcarros--;
        System.out.println("\n"+x.nome+" saiu do Estacionamento");
    }

    public void esvaziar() {
        filacarros.clear();
        numcarros = 0;
    }
}
