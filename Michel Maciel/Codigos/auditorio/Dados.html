package auditorio;

import java.util.*;

public class Dados {

    public Hashtable hashEvento;

    public Dados() {
        hashEvento = new Hashtable();
    }

    public void gravarDados(Evento meuEvento) {
        hashEvento.put(meuEvento.codEvento, meuEvento);
    }
    public void listarEventos() {
        Set<String> setKeys = hashEvento.keySet();
        for (String key : setKeys) {
            Evento xEve = (Evento) hashEvento.get(key);
            System.out.println("\t" + key + " - " + xEve.nomeEvento);
            System.out.println("\tData: " + xEve.dataEvento
                    + " / Hora: " + xEve.horaEvento+"\n");
        }
    }
    public void pesquisarEvento(String aux) {
        if (hashEvento.containsKey(aux)) {
            Evento xEve = (Evento) hashEvento.get(aux);
            System.out.println("Encontrado! - " + xEve.nomeEvento
                    + " no dia " + xEve.dataEvento + " às " + xEve.horaEvento);
        } else {
            System.out.println("Evento Não Encontrado!");
        }
    }
    public void editarEvento(String aux) {
        if (hashEvento.containsKey(aux)) {
            Evento xEve = (Evento) hashEvento.get(aux);
            System.out.println("Evento " + xEve.nomeEvento + " no dia "
                    + xEve.dataEvento + " às " + xEve.horaEvento);
            System.out.println("Digite as Alterações do Evento Abaixo\n");
            xEve.inserirEvento();
            System.out.println("\nEvento Alterado! - Novo Evento "+xEve.nomeEvento);
        } else {
            System.out.println("Evento Não Encontrado!");
        }
    }
    public void removerEvento(String aux) {

        if (hashEvento.containsKey(aux)) {
            Evento xEve = (Evento) hashEvento.get(aux);
            System.out.println("Encontrado! Evento " + xEve.nomeEvento
                    +" será Removido");
            hashEvento.remove(aux);
        } else {
            System.out.println("Evento Não Encontrado!");
        }
    }
}
