package arv;

import java.awt.AWTException;
import java.awt.Robot;
import java.awt.event.KeyEvent;

import java.util.*;

public class Arv {

    public static void main(String[] args) {
        Scanner ent = new Scanner(System.in);

        int r,num, altura;
        Arvore minhaArvore = new Arvore();

        do {
            ConsoleClear();
            System.out.println("Árvore Binária");
            System.out.println("1 - Inserir Número na Árvore");
            System.out.println("2 - Listar - EM ORDEM");
            System.out.println("3 - Listar - EM PRÉ-ORDEM");
            System.out.println("4 - Listar - EM PÓS-ORDEM");
            System.out.println("5 - Pesquisar Número na Árvore");
            System.out.println("6 - Excluir Número da Árvore");
            System.out.println("7 - Esvaziar a Árvore");
            System.out.print("Escolher Opção: ");
            r = Integer.parseInt(ent.nextLine());

            if (r == 1) {
                ConsoleClear();
                boolean status;
                System.out.print("Número - ");
                num = Integer.parseInt(ent.nextLine());
                status = minhaArvore.inserir(num);
                if (status) {
                    System.out.println("Inserido");
                } else {
                    System.out.println("Número já existe na Árvore");
                }
                ent.nextLine();
            }
            if (r == 2) {
                ConsoleClear();
                if (minhaArvore.raiz == null) {
                    System.out.println("Árvore sem Números");
                } else {
                    System.out.println("ORDEM");
                    minhaArvore.listarOrdem(minhaArvore.raiz);
                }
                ent.nextLine();
            }
            if (r == 3) {
                ConsoleClear();
                if (minhaArvore.raiz == null) {
                    System.out.println("Árvore sem Números");
                } else {
                    System.out.println("PRÉ-ORDEM");
                    minhaArvore.listarPreOrdem(minhaArvore.raiz);
                }
                ent.nextLine();
            }
            if (r == 4) {
                ConsoleClear();
                if (minhaArvore.raiz == null) {
                    System.out.println("Árvore sem Números");
                } else {
                    System.out.println("PÓS-ORDEM");
                    minhaArvore.listarPosOrdem(minhaArvore.raiz);
                }
                ent.nextLine();
            }
            if (r == 5) {
                ConsoleClear();
                System.out.print("Número - ");
                num = Integer.parseInt(ent.nextLine());
                altura = minhaArvore.pesquisar(num);
                if (altura == -1) {
                    System.out.println("Número não existe na Árvore");
                }
                System.out.println("Número existe na Árvore");
                System.out.println("Altura da Árvore: "+altura);
            }
            if (r == 6) {
                ConsoleClear();
                System.out.print("Número - ");
                num = Integer.parseInt(ent.nextLine());
                altura = minhaArvore.pesquisar(num);
                if (altura == -1) {
                    System.out.println("Número não existe na Árvore");
                } else {
                    minhaArvore.raiz = minhaArvore.excluir(minhaArvore.raiz, num);
                    System.out.println("Número excluido da Árvore");
                }
            }
            if (r == 7) {
                ConsoleClear();
                minhaArvore.raiz = null;
                System.out.println("Árvore Vazia");
            }
            ConsoleClear();
        } while (r != 0);
    }

    static void ConsoleClear() {
        try {
            Robot robot = new Robot();

            robot.setAutoDelay(5);
            robot.keyPress(KeyEvent.VK_CONTROL);
            robot.keyPress(KeyEvent.VK_L);

            robot.keyRelease(KeyEvent.VK_CONTROL);
            robot.keyRelease(KeyEvent.VK_L);

        } catch (AWTException ex) {
            System.out.println(ex.getMessage());
        }
    }
}
