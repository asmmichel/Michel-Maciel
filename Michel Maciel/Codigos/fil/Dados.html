package fil;

import java.util.*;

public class Dados {

    private List<Filme> listafilme = new ArrayList<>();

    public void inserirDados(Filme x) {
        listafilme.add(x);
    }

    public void mostrarDados() {
        for (Filme x : listafilme) {
            x.mostrarFilme();
        }
    }

    public int ordenarDados() {
        Collections.sort(listafilme, new Comparator<Filme>() {
            @Override
            public int compare(Filme a, Filme b) {
                return a.nome.compareTo(b.nome);
            }
        });
        return listafilme.size();
    }

    public Filme procurarDados(String nomeFilme) {
        Filme x = new Filme();
        x.nome = nomeFilme;
        int indice = Collections.binarySearch(listafilme, x, new Comparator<Filme>() {
            @Override
            public int compare(Filme a, Filme b) {
                return a.nome.compareTo(b.nome);
            }
        });
        if (indice >= 0) {
            x = listafilme.get(indice);
            return x;
        } else {
            return null;
        }
    }

    public void alterarDados(Filme x) {
        int indice = listafilme.indexOf(x);
        listafilme.remove(indice);
        x.inserirFilme();
        listafilme.add(indice, x);
    }
}
